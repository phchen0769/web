JavaScript简介

JavaScript是一种轻量级的脚本语言，主要用于网页交互，现在也可用于服务器端开发(Node.js)。

学习目标

掌握JavaScript基础语法、DOM操作、事件处理等核心概念，能够独立开发简单的交互式网页应用。

学习周期

建议每天学习1-2小时，完整掌握本课程内容大约需要4-6周时间。

预备知识

需要具备HTML和CSS基础，了解网页基本结构。

JavaScript基础语法

变量与数据类型

JavaScript中有三种声明变量的方式：

  • var - 函数作用域
  • let - 块级作用域（推荐）
  • const - 块级作用域，声明常量
JavaScript 代码:
// 变量声明示例
let name = "张三";
const age = 25;
var isStudent = true;

console.log("姓名:", name);
console.log("年龄:", age);
console.log("是否是学生:", isStudent);

// 尝试修改常量（会报错）
// age = 26; // TypeError: Assignment to constant variable.

流程控制

条件语句

使用if...else和switch根据条件执行不同的代码块。

JavaScript 代码:
// 条件语句示例
let score = 85;
let grade;

if (score >= 90) {
    grade = "优秀";
} else if (score >= 80) {
    grade = "良好";
} else if (score >= 70) {
    grade = "中等";
} else if (score >= 60) {
    grade = "及格";
} else {
    grade = "不及格";
}

console.log(`得分: ${score}, 等级: ${grade}`);

// switch语句
let day = new Date().getDay();
let dayName;

switch(day) {
    case 0: dayName = "星期日"; break;
    case 1: dayName = "星期一"; break;
    case 2: dayName = "星期二"; break;
    case 3: dayName = "星期三"; break;
    case 4: dayName = "星期四"; break;
    case 5: dayName = "星期五"; break;
    case 6: dayName = "星期六"; break;
    default: dayName = "未知";
}

console.log("今天是" + dayName);

函数

函数定义与调用

函数是一段可重复使用的代码块，可以接受参数并返回值。

JavaScript 代码:
// 函数定义与调用
function greet(name) {
    return "你好, " + name + "!";
}

// 函数调用
console.log(greet("张三"));
console.log(greet("李四"));

// 函数表达式
const multiply = function(a, b) {
    return a * b;
};

console.log("5 * 3 =", multiply(5, 3));

// 箭头函数 (ES6)
const divide = (a, b) => a / b;
console.log("10 / 2 =", divide(10, 2));

// 递归函数示例：计算阶乘
function factorial(n) {
    if (n === 0 || n === 1) {
        return 1;
    } else {
        return n * factorial(n - 1);
    }
}

console.log("5的阶乘:", factorial(5));

练习3: 函数编写

编写一个函数，接收一个数字参数n，返回1到n的所有数字之和。

function sumUpTo(n) {
    let sum = 0;
    for (let i = 1; i <= n; i++) {
        sum += i;
    }
    return sum;
}

// 测试函数
console.log(sumUpTo(10)); // 输出55
console.log(sumUpTo(100)); // 输出5050

DOM操作

文档对象模型(DOM)

DOM是HTML文档的编程接口，JavaScript可以通过DOM操作网页内容和样式。

JavaScript 代码:
// DOM操作示例
// 注意：在实际HTML中运行这些代码

// 获取元素
const demoElement = document.getElementById('demo');
const paragraphs = document.getElementsByTagName('p');
const items = document.querySelectorAll('.item');

// 修改内容
demoElement.innerHTML = 'Hello, JavaScript!';
demoElement.textContent = 'Hello, JavaScript!';

// 修改样式
demoElement.style.color = 'blue';
demoElement.style.fontSize = '20px';
demoElement.classList.add('highlight');

// 创建新元素
const newDiv = document.createElement('div');
newDiv.textContent = '这是一个新创建的元素';
document.body.appendChild(newDiv);

console.log("DOM操作完成!");

练习4: DOM操作

创建一个按钮，点击时改变页面上某个段落的文本内容和颜色。

// HTML部分
// <button id="changeBtn">点击改变文本</button>
// <p id="text">这是原始文本</p>

// JavaScript部分
const button = document.getElementById('changeBtn');
const textElement = document.getElementById('text');

button.addEventListener('click', function() {
    textElement.textContent = '文本已被改变!';
    textElement.style.color = 'red';
});

事件处理

事件监听与处理

JavaScript可以响应用户的各种操作，如点击、鼠标移动、键盘输入等。

JavaScript 代码:
// 事件处理示例
// 注意：在实际HTML中运行这些代码

// 添加点击事件监听器
document.getElementById('myButton').addEventListener('click', function() {
    alert('按钮被点击了!');
});

// 鼠标移入移出事件
document.getElementById('myElement').addEventListener('mouseover', function() {
    this.style.backgroundColor = 'yellow';
});

document.getElementById('myElement').addEventListener('mouseout', function() {
    this.style.backgroundColor = '';
});

// 键盘事件
document.addEventListener('keydown', function(event) {
    console.log('按下的键:', event.key);
});

console.log("事件处理示例代码已加载");

练习5: 事件处理

创建一个颜色切换器，点击按钮时切换页面的背景颜色。

// HTML部分
// <button id="colorToggle">切换背景色</button>

// JavaScript部分
const toggleButton = document.getElementById('colorToggle');
let isDark = false;

toggleButton.addEventListener('click', function() {
    if (isDark) {
        document.body.style.backgroundColor = 'white';
        document.body.style.color = 'black';
    } else {
        document.body.style.backgroundColor = '#333';
        document.body.style.color = 'white';
    }
    isDark = !isDark;
});
